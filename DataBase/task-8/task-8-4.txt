CREATE USER GAMAL IDENTIFIED BY 123;

GRANT CREATE SESSION TO GAMAL;
GRANT CREATE TABLE TO GAMAL;

CONNECT GAMAL/123;

CREATE TABLE STUDENT
(
    ID NUMBER PRIMARY KEY,
    NAME VARCHAR2(50)
);

INSERT INTO STUDENT
VALUES (1, 'Ahmed');

COMMIT;

SELECT *
FROM STUDENT;

UPDATE STUDENT
SET NAME = 'Mohamed'
WHERE ID = 1;

COMMIT;

DELETE FROM STUDENT
WHERE ID = 1;

COMMIT;

GRANT SELECT, INSERT, UPDATE, DELETE
ON STUDENT
TO HR;

INSERT INTO GAMAL.STUDENT
VALUES (2, 'Ali');

COMMIT;

SELECT *
FROM GAMAL.STUDENT;

UPDATE GAMAL.STUDENT
SET NAME = 'Omar'
WHERE ID = 2;

COMMIT;

DELETE FROM GAMAL.STUDENT
WHERE ID = 2;

COMMIT;

DELETE FROM GAMAL.STUDENT
WHERE ID = 2;

COMMIT;

REVOKE SELECT, INSERT, UPDATE, DELETE
ON STUDENT
FROM HR;

CONNECT SYS AS SYSDBA;

REVOKE CREATE TABLE FROM GAMAL;

REVOKE CREATE SESSION FROM GAMAL;

DROP USER GAMAL CASCADE;
